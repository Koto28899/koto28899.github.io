```js title='src/App.vue'
import { onMounted, ref } from 'vue'
import featuresCollection from '@/stores/featuresCollection'
import Map from '@arcgis/core/Map'
import MapView from '@arcgis/core/views/MapView'
import FeatureLayer from '@arcgis/core/layers/FeatureLayer'
import Popup from '@arcgis/core/widgets/Popup'

let featureLayer = {}

const featureLayerFields = [
  { name: 'ObjectID', alias: '点 ID', type: 'oid' },
  { name: 'type', alias: '类别', type: 'string' },
  { name: 'place', alias: '地点', type: 'string' },
  { name: 'recordedDate', alias: '记录时时间', type: 'date' },
  { name: 'imageUrl', alias: '图片链接', type: 'string' }
]

const radioGroup = [
  {
    idAndValue: 'functional-information-template',
    label: '函数型信息模板',
    onChange: (event) => {
      radioState.value = event.target.value
      featureLayer.popupTemplate = {
        title: '{ place }',
        content: () => {
          const message = featureLayerFields.map((field) => {
            return field.alias + `: {${field.name}}<br />`
          })
          return `<div>${message}</div>`
        }
      }
    }
  },
  {
    idAndValue: 'tabular-information-template',
    label: '表格型信息模板',
    onChange: (event) => {
      radioState.value = event.target.value
      featureLayer.popupTemplate = {
        title: '{ place }',
        content: [
          {
            type: 'fields',
            fieldInfos: [
              { fieldName: 'type', label: '类别' },
              { fieldName: 'place', label: '地点' },
              { fieldName: 'recordedDate', label: '记录时时间' }
            ]
          }
        ]
      }
    }
  },
  {
    idAndValue: 'text-based-information-template',
    label: '文本型信息模板',
    onChange: (event) => {
      radioState.value = event.target.value
      featureLayer.popupTemplate = {
        title: '{ place }',
        content: '地点：{place}, 类型：{type}，ID: {ObjectID}, 记录时间：{recordedDate}'
      }
    }
  },
  {
    idAndValue: 'display-image',
    label: '展示图片',
    onChange: (event) => {
      radioState.value = event.target.value
      featureLayer.popupTemplate = {
        title: '{ place }',
        content: '<img src="{ imageUrl }" alt="我手机相册里没有多少学校的建筑照，所以只有图书馆和信息学院可以展示图片"/>'
      }
    }
  }
]

const radioState = ref(radioGroup[0].idAndValue)

onMounted(() => {
  featureLayer = new FeatureLayer({
    title: 'client side point feature layer',
    fields: featureLayerFields,
    dateFieldsTimeZone: 'Asia/SHANG_HAI',
    objectIdField: 'ObjectID',
    geometryType: 'point',
    spatialReference: { wkid: 3857 },
    source: featuresCollection,
    popupTemplate: {
      title: '{ place }',
      content: [
        {
          type: 'fields',
          fieldInfos: [
            { fieldName: 'type', label: '类别' },
            { fieldName: 'place', label: '地点' },
            { fieldName: 'recordedDate', label: '记录时时间' }
          ]
        }
      ]
    },
    renderer: {
      type: 'simple',
      symbol: {
        type: 'simple-marker',
        color: [64, 158, 255],
        size: 15,
        outline: { color: [200, 200, 200], width: 1.5 }
      }
    },
    labelsVisible: false
  })

  const map = new Map({
    basemap: 'arcgis/imagery',
    layers: [featureLayer]
  })

  const popup = new Popup({
    visibleElements: {
      collapseButton: false,
      actionBar: false,
      closeButton: false
    },
    dockOptions: {
      buttonEnabled: false
    }
  })

  const mapView = new MapView({
    container: 'viewDiv',
    map: map,
    center: [121.89255062207127, 30.885831068414372],
    zoom: 17,
    popup: popup
  })
})
```
